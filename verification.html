$var[captcha;true]        $c[Type "false" to verify the member directly on click, no captcha]
 
  $var[role;Role ID]        $c[Role ID of verified/unverified role]
   
  $var[type;+]              $c[Type "-" to remove the role ID mentioned above instead of adding upon verification]
   
  $var[case-sensitive;true] $c[Set "false" to make the bot accept any cases.]
   
  $nomention
  $onlyIf[$checkContains[$customID;userverification]==true;]
  $ephemeral
  $if[$checkContains[$customID;userverification.start]==true]
  $if[$var[captcha]==true]
  $var[text;$randomString[5]]
  $removeButtons
  $color[#303136]
  $image[https://flamingtext.com/net-fu/proxy_form.cgi?script=crafts-logo&text=++++++$var[text]++++++&_loc=generate&imageoutput=true&backgroundRadio=1&backgroundColor=%23303136]
  $addButton[yes;userverification.modal.$var[text];Enter Captcha;success;no;]
  $addButton[no;userverification.start;Generate New Captcha;secondary;no;]
  $else
  $removeButtons
  $roleGrant[$authorID;$var[type]$var[role]]
  ✅ You are Verified.
  $endif
  $elseif[$checkContains[$customID;userverification.modal]==true]
  $textSplit[$customID;.]
  $newModal[userverification.input.$splitText[3];Captcha]
  $addTextInput[cap;short;Enter Captcha Here;5;5;yes;;]
  $elseif[$checkContains[$customID;userverification.input]==true]
  $textSplit[$customID;.]
  $if[$var[case-sensitive]==false]
  $var[input;$toLowercase[$input[cap]]]
  $var[capth;$toLowercase[$splitText[3]]]
  $else
  $var[input;$input[cap]]
  $var[capth;$splitText[3]]
  $endif
  $if[$var[input]==$var[capth]]
  $roleGrant[$authorID;$var[type]$var[role]]
  ✅ You are Verified.
  $else
  ❗ Wrong Captcha! 
  $endif
  $endif
